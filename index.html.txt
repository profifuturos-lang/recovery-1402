<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>WhatsApp Task Manager - Standalone</title>
    <!-- React & ReactDOM via CDN -->
    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    <!-- Babel para procesar JSX en el navegador -->
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <!-- Tailwind CSS via CDN -->
    <script src="https://cdn.tailwindcss.com"></script>
    <link href="https://fonts.googleapis.com/css2?family=Inter:wght@400;600;700&display=swap" rel="stylesheet">
    <style>
        body { font-family: 'Inter', sans-serif; background-color: #0a0a0a; color: #f4f4f5; margin: 0; }
        .glass { background: rgba(24, 24, 27, 0.8); backdrop-filter: blur(12px); border: 1px solid rgba(39, 39, 42, 0.5); }
        ::-webkit-scrollbar { width: 8px; }
        ::-webkit-scrollbar-track { background: #0a0a0a; }
        ::-webkit-scrollbar-thumb { background: #27272a; border-radius: 10px; }
    </style>
</head>
<body>
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useEffect } = React;

        // Iconos SVG (Reemplazo de Lucide para independencia total)
        const Icons = {
            MessageSquare: () => <svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><path d="M21 15a2 2 0 0 1-2 2H7l-4 4V5a2 2 0 0 1 2-2h14a2 2 0 0 1 2 2z"/></svg>,
            Plus: () => <svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><line x1="12" y1="5" x2="12" y2="19"/><line x1="5" y1="12" x2="19" y2="12"/></svg>,
            LogIn: () => <svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><path d="M15 3h4a2 2 0 0 1 2 2v14a2 2 0 0 1-2 2h-4"/><polyline points="10 17 15 12 10 7"/><line x1="15" y1="12" x2="3" y2="12"/></svg>,
            X: () => <svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"><line x1="18" y1="6" x2="6" y2="18"/><line x1="6" y1="6" x2="18" y2="18"/></svg>,
            Play: () => <svg width="16" height="16" viewBox="0 0 24 24" fill="currentColor"><polygon points="5 3 19 12 5 21 5 3"/></svg>,
            Trash2: () => <svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><polyline points="3 6 5 6 21 6"/><path d="M19 6v14a2 2 0 0 1-2 2H7a2 2 0 0 1-2-2V6m3 0V4a2 2 0 0 1 2-2h4a2 2 0 0 1 2 2v2"/><line x1="10" y1="11" x2="10" y2="17"/><line x1="14" y1="11" x2="14" y2="17"/></svg>,
            CheckCircle: () => <svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2"><path d="M22 11.08V12a10 10 0 1 1-5.93-9.14"/><polyline points="22 4 12 14.01 9 11.01"/></svg>
        };

        function App() {
            const [tasks, setTasks] = useState([]);
            const [isAdminOpen, setIsAdminOpen] = useState(false);
            const [isAuthenticated, setIsAuthenticated] = useState(false);
            const [adminPassword, setAdminPassword] = useState('');
            const [phone, setPhone] = useState('');
            const [message, setMessage] = useState('');

            // Cargar de LocalStorage
            useEffect(() => {
                const saved = localStorage.getItem('wa_tasks_standalone');
                if (saved) setTasks(JSON.parse(saved));
            }, []);

            // Guardar en LocalStorage
            useEffect(() => {
                localStorage.setItem('wa_tasks_standalone', JSON.stringify(tasks));
            }, [tasks]);

            const handleLogin = (e) => {
                e.preventDefault();
                if (adminPassword === 'PERRA') { setIsAuthenticated(true); setAdminPassword(''); }
                else alert('Contraseña incorrecta');
            };

            const addTask = (e) => {
                e.preventDefault();
                const newTask = { id: Date.now().toString(), phone: phone.replace(/\D/g, ''), message, status: 'DISPONIBLE' };
                setTasks([newTask, ...tasks]);
                setPhone(''); setMessage('');
            };

            const executeTask = (task) => {
                window.open(`https://wa.me/${task.phone}?text=${encodeURIComponent(task.message)}`, '_blank');
                setTasks(tasks.map(t => t.id === task.id ? { ...t, status: 'COMPLETADO' } : t));
            };

            const deleteTask = (id) => setTasks(tasks.filter(t => t.id !== id));

            return (
                <div className="min-h-screen pb-20">
                    <header className="sticky top-0 z-50 border-b border-zinc-800/50 bg-black/80 backdrop-blur-md h-16 flex items-center justify-between px-4 max-w-5xl mx-auto w-full">
                        <div className="flex items-center gap-2">
                            <div className="w-8 h-8 bg-emerald-500 rounded-lg flex items-center justify-center"><Icons.MessageSquare /></div>
                            <h1 className="text-lg font-bold">WA Task Manager</h1>
                        </div>
                        <button onClick={() => setIsAdminOpen(!isAdminOpen)} className="flex items-center gap-2 px-4 py-2 rounded-full bg-zinc-900 border border-zinc-800 text-sm">
                            {isAdminOpen ? <Icons.X /> : <Icons.LogIn />} {isAdminOpen ? 'Cerrar' : 'Admin'}
                        </button>
                    </header>

                    <main className="max-w-5xl mx-auto px-4 py-8">
                        {isAdminOpen && (
                            <div className="mb-12 p-6 rounded-2xl glass shadow-2xl">
                                {!isAuthenticated ? (
                                    <form onSubmit={handleLogin} className="max-w-sm mx-auto text-center space-y-4 py-4">
                                        <h2 className="text-lg font-semibold">Acceso Admin</h2>
                                        <input type="password" value={adminPassword} onChange={e => setAdminPassword(e.target.value)} placeholder="Contraseña" className="w-full bg-black border border-zinc-800 rounded-xl px-4 py-2.5 outline-none text-center" required />
                                        <button className="w-full bg-white text-black font-bold py-2.5 rounded-xl">Entrar</button>
                                    </form>
                                ) : (
                                    <form onSubmit={addTask} className="grid grid-cols-1 md:grid-cols-2 gap-4">
                                        <input value={phone} onChange={e => setPhone(e.target.value)} placeholder="Número" className="bg-black border border-zinc-800 rounded-xl px-4 py-3 outline-none" required />
                                        <textarea value={message} onChange={e => setMessage(e.target.value)} placeholder="Mensaje" className="bg-black border border-zinc-800 rounded-xl px-4 py-3 outline-none h-12 resize-none" required />
                                        <button className="md:col-span-2 bg-emerald-500 text-black font-bold py-3 rounded-xl flex items-center justify-center gap-2"><Icons.Plus /> Agregar Tarea</button>
                                    </form>
                                )}
                            </div>
                        )}

                        <div className="space-y-12">
                            <Section title="Disponibles" count={tasks.filter(t => t.status === 'DISPONIBLE').length} color="emerald">
                                {tasks.filter(t => t.status === 'DISPONIBLE').map(t => <TaskCard key={t.id} task={t} onExecute={executeTask} onDelete={deleteTask} isAdmin={isAuthenticated} />)}
                            </Section>
                            <Section title="Completadas" count={tasks.filter(t => t.status === 'COMPLETADO').length} color="zinc">
                                {tasks.filter(t => t.status === 'COMPLETADO').map(t => <TaskCard key={t.id} task={t} onExecute={executeTask} onDelete={deleteTask} isAdmin={isAuthenticated} isDone />)}
                            </Section>
                        </div>
                    </main>
                </div>
            );
        }

        function Section({ title, count, color, children }) {
            return (
                <div>
                    <div className="flex items-center justify-between mb-6">
                        <h2 className={`text-xl font-bold ${color === 'zinc' ? 'text-zinc-500' : ''}`}>{title}</h2>
                        <span className={`text-[10px] font-bold px-2 py-1 rounded bg-emerald-500/10 text-emerald-500`}>{count}</span>
                    </div>
                    <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">{count === 0 ? <p className="col-span-full text-zinc-700 text-center py-8 border border-dashed border-zinc-800 rounded-xl">Vacío</p> : children}</div>
                </div>
            );
        }

        function TaskCard({ task, onExecute, onDelete, isAdmin, isDone }) {
            return (
                <div className={`group relative p-5 rounded-2xl border transition-all ${isDone ? 'border-zinc-900 bg-zinc-900/10 opacity-50 grayscale' : 'border-zinc-800 bg-zinc-900/40 hover:border-emerald-500/30'}`}>
                    <div className="flex justify-between items-start mb-4">
                        <span className={`text-[10px] font-bold px-2 py-0.5 rounded-full ${isDone ? 'bg-zinc-800 text-zinc-500' : 'bg-emerald-500/10 text-emerald-500'}`}>{isDone ? 'COMPLETADO' : 'LISTO'}</span>
                        {isAdmin && <button onClick={() => onDelete(task.id)} className="text-zinc-700 hover:text-red-500 opacity-0 group-hover:opacity-100"><Icons.Trash2 /></button>}
                    </div>
                    <button onClick={() => onExecute(task)} className={`w-full py-4 rounded-xl font-bold text-xs flex items-center justify-center gap-2 ${isDone ? 'bg-zinc-800 text-zinc-500' : 'bg-emerald-500 text-black shadow-lg shadow-emerald-500/10'}`}>
                        {isDone ? <Icons.CheckCircle /> : <Icons.Play />} {isDone ? 'ENVIADO' : 'CLICK PARA ENVIAR MENSAJE'}
                    </button>
                </div>
            );
        }

        ReactDOM.createRoot(document.getElementById('root')).render(<App />);
    </script>
</body>
</html>
